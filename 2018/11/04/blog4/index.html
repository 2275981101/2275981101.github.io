<style>
    .myCheck{
        display: none;
    }
    .myleb{
        display: inline-block;
        width: 16px;
        height: 16px;
        position: relative;
        border: 1px solid #abcdef;
        border-radius: 15%;
        margin-right: 3px;
        vertical-align: middle;
    }
    .myleb::before{
        content: "";
        width: 10px;
        height: 10px;
        position: absolute;
        top: 3px;
        left: 3px;
        background-color: #abcdef;
        border-radius: 15%;
        display: none;
    }
    .myCheck:checked+.myleb::before{
        display: block;
    }
</style>
